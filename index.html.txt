    <script>
        var globalDataDesa = {};
        var globalJabatanMap = {}; 

        // Fungsi mengganti Judul & memanggil perubahan Opsi Dropdown
        function updateSectionTitle(selectedValue) {
            var titleElement = document.getElementById('dynamic-title');
            titleElement.innerHTML = `<i class="fa-solid fa-id-badge" style="margin-right: 8px;"></i> 1. Identitas Pribadi - ${selectedValue}`;
            updateJabatanDropdown(selectedValue);
        }

        // FUNGSI INTI: Mengganti opsi Dropdown Jabatan sesuai Kategori Saklar
        function updateJabatanDropdown(kategori) {
            var jabSelect = document.getElementById('jabatan');
            jabSelect.innerHTML = '<option value="">-- Pilih Jabatan --</option>';

            // globalJabatanMap sudah memetakan pilihan "PPK", "PPS", dll ke daftarnya
            if (globalJabatanMap[kategori]) {
                var listJabatan = globalJabatanMap[kategori];
                listJabatan.forEach(function(jab) {
                    // Filter agar baris kosong dari Google Sheet tidak ikut masuk
                    if (jab && jab.trim() !== "") {
                        jabSelect.add(new Option(jab, jab));
                    }
                });
            }
        }

        document.addEventListener("DOMContentLoaded", function() {
            loadDropdownData();
        });

        // Mengambil seluruh data dari Apps Script (Kecamatan, Desa, dan ke-4 Kolom Jabatan)
        function loadDropdownData() {
            google.script.run.withSuccessHandler(function(response) {
                if(response.error) {
                    console.error("Gagal memuat data dropdown: ", response.error);
                    document.getElementById('kecamatan').innerHTML = '<option value="">Error memuat data</option>';
                    document.getElementById('jabatan').innerHTML = '<option value="">Error memuat data</option>';
                    return;
                }
                
                // Simpan Map Data ke variabel global
                globalJabatanMap = response.jabatanMap;
                globalDataDesa = response.desaMap;

                // Terapkan opsi jabatan otomatis untuk pilihan saklar pertama (biasanya "PPK" yang aktif duluan)
                var kategoriAwal = document.querySelector('input[name="kategori_karyawan"]:checked').value;
                updateJabatanDropdown(kategoriAwal);

                // Set Dropdown Kecamatan
                var kecSelect = document.getElementById('kecamatan');
                kecSelect.innerHTML = '<option value="">-- Pilih Kecamatan --</option>';
                response.kecamatan.forEach(function(kec) {
                    if (kec && kec.trim() !== "") {
                        kecSelect.add(new Option(kec, kec));
                    }
                });
            }).getDropdownDataFromSheet();
        }

        // Mengubah desa sesuai kecamatan yang dipilih
        function updateDesaDropdown() {
            var kecTerpilih = document.getElementById('kecamatan').value;
            var desaSelect = document.getElementById('kelurahan');
            desaSelect.innerHTML = '<option value="">-- Pilih Kelurahan/Desa --</option>';
            if (kecTerpilih && globalDataDesa[kecTerpilih]) {
                var listDesa = globalDataDesa[kecTerpilih];
                listDesa.sort();
                listDesa.forEach(function(desa) {
                    if (desa && desa.trim() !== "") {
                        desaSelect.add(new Option(desa, desa));
                    }
                });
            }
        }

        function switchView(viewName) {
            var sections = document.querySelectorAll('.view-section');
            sections.forEach(function(sec) { sec.classList.remove('active-view'); });
            var menuMatrik = document.getElementById('menu-matrik');
            var menuInput = document.getElementById('menu-input');
            menuMatrik.classList.remove('active');
            menuInput.classList.remove('active');
            if (viewName === 'matrik') {
                document.getElementById('view-matrik').classList.add('active-view');
                menuMatrik.classList.add('active');
                fetchLiveDashboardData(); 
            } else {
                document.getElementById('view-input').classList.add('active-view');
                menuInput.classList.add('active');
            }
        }

        function prosesSimpan() {
            var btn = document.getElementById('btnSubmit');
            var status = document.getElementById('statusMsg');
            
            btn.disabled = true;
            btn.innerHTML = '<i class="fa-solid fa-spinner fa-spin"></i> Menyimpan...';
            status.innerText = "";

            var kategoriTerpilih = document.querySelector('input[name="kategori_karyawan"]:checked').value;

            var data = {
                kategoriKaryawan: kategoriTerpilih,
                nonik: document.getElementById('nonik').value,
                namaLengkap: document.getElementById('namaLengkap').value,
                namaSingkat: document.getElementById('namaSingkat').value,
                jabatan: document.getElementById('jabatan').value,
                tanggalLahir: document.getElementById('tanggalLahir').value,
                tempatLahir: document.getElementById('tempatLahir').value,
                namaIbu: document.getElementById('namaIbu').value,
                agama: document.getElementById('agama').value,
                statusPernikahan: document.getElementById('statusPernikahan').value,
                pendidikan: document.getElementById('pendidikan').value,
                jenisKelamin: document.getElementById('jenisKelamin').value,
                alamat: document.getElementById('alamat').value,
                rt: document.getElementById('rt').value,
                rw: document.getElementById('rw').value,
                kelurahan: document.getElementById('kelurahan').value,
                kecamatan: document.getElementById('kecamatan').value,
                kodePos: document.getElementById('kodePos').value,
                noHp: document.getElementById('noHp').value
            };

            google.script.run.withSuccessHandler(function(response) {
                status.innerText = response;
                status.style.color = response.includes("Gagal") ? "var(--danger)" : "var(--success)";
                if(!response.includes("Gagal")) {
                    document.getElementById('dataForm').reset();
                    document.getElementById('kelurahan').innerHTML = '<option value="">-- Pilih Kecamatan Terlebih Dahulu --</option>';
                    
                    // Kembalikan status saklar ke PPK
                    var kategoriAwal = "PPK";
                    document.getElementById('opt-ppk').checked = true;
                    updateSectionTitle(kategoriAwal);
                }
                btn.disabled = false;
                btn.innerHTML = '<i class="fa-solid fa-floppy-disk"></i> Simpan Data';
                setTimeout(function(){ status.innerText = ""; }, 5000);
            }).simpanData(data);
        }

        function fetchLiveDashboardData() {
            google.script.run.withSuccessHandler(function(data) {
                if(data.error) {
                    console.error("Error spreadsheet data fetch: ", data.error);
                    return;
                }

                document.getElementById('card-total-karyawan').innerText = data.totalKaryawan;
                document.getElementById('card-sudah-mengisi').innerText = data.sudahMengisi;
                document.getElementById('card-belum-mengisi').innerText = data.belumMengisi;
                document.getElementById('card-jumlah-kecamatan').innerText = data.jumlahKecamatan;
                document.getElementById('card-jumlah-desa').innerText = data.jumlahDesa;

                var pSudah = data.totalKaryawan > 0 ? ((data.sudahMengisi / data.totalKaryawan)*100).toFixed(2) : "0.00";
                var pBelum = data.totalKaryawan > 0 ? ((data.belumMengisi / data.totalKaryawan)*100).toFixed(2) : "0.00";
                document.getElementById('pct-sudah-mengisi').innerText = pSudah + "% Target Tercapai";
                document.getElementById('pct-belum-mengisi').innerText = pBelum + "% Belum Selesai";

                var tableMainBody = document.getElementById('live-main-table');
                tableMainBody.innerHTML = "";
                data.kecamatanData.forEach(function(item, index) {
                    var badgeStyle = "badge-kurang";
                    if(item.status === "Sangat Baik") badgeStyle = "badge-sangat-baik";
                    else if(item.status === "Baik") badgeStyle = "badge-baik";
                    else if(item.status === "Cukup") badgeStyle = "badge-cukup";

                    var row = `<tr>
                        <td>${index + 1}</td>
                        <td style="text-align:left; font-weight:600; color:var(--text-main);">${item.kecamatan}</td>
                        <td>${item.jumlahDesa}</td>
                        <td>${item.totalKaryawan}</td>
                        <td style="font-weight:700; color:var(--success);">${item.sudahJumlah}</td>
                        <td style="font-weight:600;">${item.sudahPersen}</td>
                        <td style="font-weight:600; color:var(--danger);">${item.belumJumlah}</td>
                        <td>${item.belumPersen}</td>
                        <td style="width:130px; text-align:left;">
                            <span style="font-size:11px; font-weight:600;">${item.sudahPersen}</span>
                            <div class="progress-wrapper"><div class="progress-fill" style="width:${item.sudahPersen};"></div></div>
                        </td>
                        <td><span class="badge ${badgeStyle}">${item.status}</span></td>
                        <td><button class="btn-detail"><i class="fa-solid fa-list-ul"></i> Detail</button></td>
                    </tr>`;
                    tableMainBody.innerHTML += row;
                });
                document.getElementById('table-info-text').innerText = `Menampilkan 1 - ${data.kecamatanData.length} dari total ${data.kecamatanData.length} kecamatan`;

                var tableUsiaBody = document.getElementById('live-usia-table');
                tableUsiaBody.innerHTML = "";
                data.usiaData.forEach(function(item) {
                    var pctNum = parseFloat(item.persen);
                    var row = `<tr>
                        <td style="text-align: left; padding-left: 0; font-weight: 600;">${item.kelompok}</td>
                        <td>${item.jumlah}</td>
                        <td style="font-weight: 600;">${item.persen}</td>
                        <td style="text-align: left;"><div class="age-bar" style="width:${Math.max(pctNum, 5)}%;"></div></td>
                    </tr>`;
                    tableUsiaBody.innerHTML += row;
                });

                var barContainer = document.getElementById('live-bar-chart');
                barContainer.innerHTML = '<div class="y-axis"><span>50</span><span>40</span><span>30</span><span>20</span><span>10</span><span>0</span></div>';
                
                data.kecamatanData.forEach(function(item) {
                    var heightPercent = Math.min(100, (item.sudahJumlah / 50) * 100);
                    var barGroup = document.createElement('div');
                    barGroup.className = 'bar-group';
                    barGroup.innerHTML = `
                        <div class="bar" style="height: ${heightPercent}%;">
                            <span class="bar-value">${item.sudahJumlah > 0 ? item.sudahJumlah : ''}</span>
                        </div>
                        <span class="bar-label">${item.kecamatan}</span>
                    `;
                    barContainer.appendChild(barGroup);
                });

            }).getDashboardData();
        }

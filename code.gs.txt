// Fungsi Mengambil Data Dropdown dari 2 Sheet Berbeda
function getDropdownDataFromSheet() {
  try {
    var SPREADSHEET_ID = "13bTw9rTXPfdaHeTnCeX-NrpqG8hj5VR5AZUZCsNlYwY";
    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    
    // 1. Ambil Data Kecamatan dan Desa
    var sheetDesa = ss.getSheetByName("data_des");
    if(!sheetDesa) throw new Error("Sheet 'data_des' tidak ditemukan.");
    var dataDesa = sheetDesa.getDataRange().getValues();
    
    var kecSet = {};
    var desaMap = {};
    var currentKecamatan = ""; // Variabel penampung ingatan kecamatan
    
    for(var i = 1; i < dataDesa.length; i++) {
      var row = dataDesa[i];
      var kecRaw = row[0] ? row[0].toString().trim() : ""; // Asumsi Kolom A: Kecamatan
      var desa = row[1] ? row[1].toString().trim() : "";   // Asumsi Kolom B: Desa

      // Jika kolom kecamatan tidak kosong, update ingatan kita
      if(kecRaw !== "") {
        currentKecamatan = kecRaw;
        kecSet[currentKecamatan] = true;
      }

      // Masukkan desa ke dalam daftar kecamatan yang sedang diingat
      if(currentKecamatan !== "") {
        if(!desaMap[currentKecamatan]) desaMap[currentKecamatan] = [];
        if(desa && desaMap[currentKecamatan].indexOf(desa) === -1) {
          desaMap[currentKecamatan].push(desa);
        }
      }
    }

    // 2. Ambil Data Jabatan
    var sheetJabatan = ss.getSheetByName("jabatan");
    if(!sheetJabatan) throw new Error("Sheet 'jabatan' tidak ditemukan.");
    var dataJabatan = sheetJabatan.getDataRange().getValues();

    var jabPPK = {}, jabSekPPK = {}, jabPPS = {}, jabSekPPS = {};

    for(var j = 1; j < dataJabatan.length; j++) {
      var rowJ = dataJabatan[j];
      var valPPK = rowJ[0] ? rowJ[0].toString().trim() : "";    // Kolom A
      var valSekPPK = rowJ[1] ? rowJ[1].toString().trim() : ""; // Kolom B
      var valPPS = rowJ[2] ? rowJ[2].toString().trim() : "";    // Kolom C
      var valSekPPS = rowJ[3] ? rowJ[3].toString().trim() : ""; // Kolom D

      if(valPPK) jabPPK[valPPK] = true;
      if(valSekPPK) jabSekPPK[valSekPPK] = true;
      if(valPPS) jabPPS[valPPS] = true;
      if(valSekPPS) jabSekPPS[valSekPPS] = true;
    }

    return {
      kecamatan: Object.keys(kecSet).sort(),
      desaMap: desaMap,
      jabatanMap: {
        "PPK": Object.keys(jabPPK),
        "Sekretariat PPK": Object.keys(jabSekPPK),
        "PPS": Object.keys(jabPPS),
        "Sekretariat PPS": Object.keys(jabSekPPS)
      }
    };

  } catch(e) {
    return { error: e.toString() };
  }
}

function getDashboardData() {
  try {
    var SPREADSHEET_ID = "13bTw9rTXPfdaHeTnCeX-NrpqG8hj5VR5AZUZCsNlYwY";
    var ss = SpreadsheetApp.openById(SPREADSHEET_ID);
    var sheet = ss.getSheetByName("Data_PPK");
    if(!sheet) return { error: "Sheet Data_PPK kosong" };
    var data = sheet.getDataRange().getValues();
    
    var targetKecamatan = {
      "KEC. A": { target: 45, desa: 18 }, "KEC. B": { target: 32, desa: 16 },
      "KEC. C": { target: 28, desa: 17 }, "KEC. D": { target: 21, desa: 14 },
      "KEC. E": { target: 18, desa: 12 }, "KEC. F": { target: 17, desa: 13 },
      "KEC. G": { target: 15, desa: 11 }
    };
    
    var counts = {};
    var usiaGroups = { "&le; 25 tahun": 0, "26 – 30 tahun": 0, "31 – 35 tahun": 0, "36 – 40 tahun": 0, "41 – 45 tahun": 0, "46 – 50 tahun": 0, "> 50 tahun": 0 };
    var sudahMengisiCount = 0;

    if (data.length > 1) {
      var headers = data[0];
      var rows = data.slice(1);
      sudahMengisiCount = rows.length;
      
      var idxKecamatan = headers.indexOf("Kecamatan");
      var idxTglLahir = headers.indexOf("Tanggal Lahir");
      rows.forEach(function(row) {
        var kecRaw = row[idxKecamatan] ? row[idxKecamatan].toString().trim().toUpperCase() : "LAINNYA";
        counts[kecRaw] = (counts[kecRaw] || 0) + 1;
        
        var tglLahirStr = row[idxTglLahir];
        if (tglLahirStr) {
          var birthDate = new Date(tglLahirStr);
          if (!isNaN(birthDate.getTime())) {
            var age = new Date().getFullYear() - birthDate.getFullYear();
            if (age <= 25) usiaGroups["&le; 25 tahun"]++;
            else if (age <= 30) usiaGroups["26 – 30 tahun"]++;
            else if (age <= 35) usiaGroups["31 – 35 tahun"]++;
            else if (age <= 40) usiaGroups["36 – 40 tahun"]++;
            else if (age <= 45) usiaGroups["41 – 45 tahun"]++;
            else if (age <= 50) usiaGroups["46 – 50 tahun"]++;
            else usiaGroups["> 50 tahun"]++;
          }
        }
      });
    }

    var kecamatanList = [];
    var allKecamatans = Object.keys(targetKecamatan);
    Object.keys(counts).forEach(function(k) { if (allKecamatans.indexOf(k) === -1) allKecamatans.push(k); });
    var totalKaryawanSemua = 0;
    var totalDesaSemua = 0;

    allKecamatans.forEach(function(kec) {
      var sudah = counts[kec] || 0;
      var info = targetKecamatan[kec] || { target: sudah, desa: 1 };
      var total = Math.max(info.target, sudah);
      var belum = Math.max(0, total - sudah);
      
      totalKaryawanSemua += total;
      totalDesaSemua += info.desa;
      
      var pctSudah = total > 0 ? ((sudah / total) * 100).toFixed(2) : "0.00";
      var pctBelum = total > 0 ? ((belum / total) * 100).toFixed(2) : "0.00";
      
      var status = "Kurang";
      var numPct = parseFloat(pctSudah);
      if (numPct >= 90) status = "Sangat Baik";
      else if (numPct >= 75) status = "Baik";
      else if (numPct >= 50) status = "Cukup";

      kecamatanList.push({
        kecamatan: kec, jumlahDesa: info.desa, totalKaryawan: total,
        sudahJumlah: sudah, sudahPersen: pctSudah + "%", belumJumlah: belum, belumPersen: pctBelum + "%", status: status
      });
    });

    kecamatanList.sort(function(a, b) { return a.kecamatan.localeCompare(b.kecamatan); });
    var belumMengisiCount = totalKaryawanSemua - sudahMengisiCount;

    return {
      totalKaryawan: totalKaryawanSemua, sudahMengisi: sudahMengisiCount, belumMengisi: belumMengisiCount,
      jumlahKecamatan: allKecamatans.length, jumlahDesa: totalDesaSemua, kecamatanData: kecamatanList,
      usiaData: Object.keys(usiaGroups).map(function(key) {
        return { kelompok: key, jumlah: usiaGroups[key], persen: sudahMengisiCount > 0 ? ((usiaGroups[key] / sudahMengisiCount) * 100).toFixed(2) + "%" : "0.00%" };
      })
    };
  } catch(e) {
    return { error: e.toString() };
  }
}
